! Cisco NX-OS NetFlow (v9/IPFIX) example
feature netflow
flow record ELASTIC-FLOW
  match ipv4 source address
  match ipv4 destination address
  match ip protocol
  match transport source-port
  match transport destination-port
  collect counter bytes long
  collect counter packets long
  collect timestamp sys-uptime first
  collect timestamp sys-uptime last
!
flow exporter ELASTIC-EXP
  destination <COLLECTOR_IP>
  transport udp 2055
  version 9
!
flow monitor ELASTIC-MON
  record ELASTIC-FLOW
  exporter ELASTIC-EXP
!
interface <INTERFACE>
  ip flow monitor ELASTIC-MON input
  ip flow monitor ELASTIC-MON output
